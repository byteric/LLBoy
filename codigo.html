import serial
import time
import winsound 

porta = 'COM7' 
baud_rate = 9600
ultima_contagem = None

def bip(frequencia, duracao):
    winsound.Beep(frequencia, duracao)

def explosao():
    
    winsound.Beep(1100, 5000)
    time.sleep(0.1)

    

try:
    arduino = serial.Serial(porta, baud_rate, timeout=1)
    time.sleep(2)
    print(f"Lendo da porta {porta}...\n")

    while True:
        if arduino.in_waiting > 0:
            linha = arduino.readline().decode('utf-8').strip()
            print(linha)

            if "tempo restante" in linha:
                partes = linha.split()
                tempo = int(partes[-1])

                if tempo != ultima_contagem:
                    ultima_contagem = tempo

                    if tempo > 20:
                        bip(1000, 200) 
                    elif tempo > 10:
                        bip(1000, 300) 
                    elif tempo > 0:
                        bip(1000, 400) 
                    elif tempo == 0:
                        explosao()

except KeyboardInterrupt:
    print("\nEncerrado pelo usu谩rio.")
finally:
    if 'arduino' in locals() and arduino.is_open:
        arduino.close()
